git clone https://github.com/jaemyoun/YCSB.git
cd YSCB/
git checkout 7aaec27

# Build
mvn clean package

# Compile only RADOS-binding
mvn -pl com.yahoo.ycsb:rados-binding -am clean package

# You can set configurations with the shell command.
./bin/ycsb.sh load rados -s -P workloads/workloada -p "rados.configfile=/etc/ceph/ceph.conf" -p "rados.id=admin" -p "rados.pool=data" > outputLoad.txt
./bin/ycsb.sh run rados -s -P workloads/workloada > outputRun.txt



##### Following cases are examples of checking workloads in various databases.
# CASE: Cassandra database (Reference: http://call518.tistory.com/116)
./bin/ycsb load cassandra-10 -P workloads/workloada -p hosts=172.21.81.139,172.21.81.127 -p cassandra.connectionretries=1 -p cassandra.operationretries=1

# CASE: MongoDB (Reference: https://scalegrid.io/blog/how-to-benchmark-mongodb-with-ycsb/)
./bin/ycsb load mongodb -s -P workloads/workloada -p recordcount=10000000 -threads 16 -p
 mongodb.url="mongodb://user:pwd@server1.example.com:9999,server2.example.com:9999/dbname" -p 
 mongodb.auth="true"
